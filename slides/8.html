let message = Currency`You should pay ${value} @currency@ discount.`;

function Currency(templateData, ...args){
    return templateData.map((part, index) => {
        let str = part;
        if( args[index] ){
            str = part + String(args[index]);
        }
        return str.replace('@currency@', 'PLN');
    }).join('');
}
